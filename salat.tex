\section{Salate}

\begin{recipe}{Glasnudelsalat Shanghai}
  \ingredient{110\,g Glasnudeln}
  nach Vorschrift kochen.

  \ingredient{250\,g Hühner- oder Putenfleisch}
  kochen, klein schneiden.

  \ingredient{2\,EL Speiseöl}
  \ingredient{2\,EL Essig}
  \ingredient{2\,EL Sojasoße}
  \ingredient{1–2\,EL Currypulver}
  \ingredient{1\,Msp. Sambal Oelek}
  verrühren, mit den Glasnudeln, dem Fleisch sowie
  \ingredient{280\,g Erbsen (Konserve)}
  \ingredient{200\,g Mandarinen (Konserve)}
  \ingredient{170\,g Soja- oder Mungbohnenkeimlinge (Konserve)}
  \ingredient{225\,g Bambussprossen (Konserve)}
  \ingredient{1 Stck. eingelegter Ingwer oder Ingwerpulver}
  vermengen.

  Gut durchziehen lassen (möglichst über Nacht), evtl. mit Currypulver
  und Sambal Oelek abschmecken.
\end{recipe}
