\section{Cremes}

\begin{recipe}{Eiscreme}
  \ingredient{10 Eigelb}
  \ingredient{10\,EL Zucker}
  schaumig rühren.

  \ingredient{500\,ml Sahne}
  schlagen.
  
  \ingredient{10 Eiweiß}
  zu Eischnee schlagen.

  \ingredient{12\,Bl. weiße Gelatine}
  in Wasser einweichen, ausdrücken, in
  \ingredient[Rum]{125\,ml Rum}
  auflösen (im Wasserbad) und zur Zucker-Ei-Masse geben.

  Sobald die Masse steif wird, Eischnee und geschlagene Sahne
  unterheben.

  Die Creme schichtweise in Schalen füllen, zwischen die Schichten
  \ingredient[Schokolade]{100\,g Raspelschokolade}
  geben. Mit Schokoladenstreusseln verzieren.
\end{recipe}

\begin{recipe}{Mohrenkopf}
  \ingredient{250\,ml Milch}
  \ingredient{2\,Pkg. Vanillezucker}
  erhitzen, dann kalt stellen.

  \ingredient{5 Eigelb}
  \ingredient{100\,g Zucker}
  schaumig rühren, gezuckerte Milch dazugeben.

  \ingredient{250\,ml Sahne}
  schlagen.
  
  \ingredient{5 Eiweiß}
  zu Eischnee schlagen.

  \ingredient{8\,Bl. weiße Gelatine}
  in wenig Wasser  auflösen (im Wasserbad) und zur Zucker-Ei-Masse
  geben.

  Sobald die Masse steif wird, Eischnee und geschlagene Sahne
  unterheben. 3 Schalen zur Hälfte füllen.

  \ingredient{250\,ml Milch}
  \ingredient[Kakao]{2\,EL Kakao}
  erhitzen,
  \ingredient[Schokolade]{100\,g Schokolade}
  darin auflösen, dann kalt stellen.

  \ingredient{5 Eigelb}
  \ingredient{100\,g Zucker}
  schaumig rühren, Schokomilch dazugeben.

  \ingredient{250\,ml Sahne}
  schlagen.
  
  \ingredient{5 Eiweiß}
  zu Eischnee schlagen.

  \ingredient{8\,Bl. weiße Gelatine}
  in wenig Wasser  auflösen (im Wasserbad) und zur Zucker-Ei-Masse
  geben.

  Sobald die Masse steif wird, Eischnee und geschlagene Sahne
  unterheben. Die Schokoladencreme auf die helle Creme füllen.
\end{recipe}

\begin{recipe}{Welfenspeise}
  \ingredient{3 Eiweiß}
  steif schlagen.
  
  \ingredient{300\,ml Milch}
  \ingredient{50\,g Zucker}
  \ingredient{1\,TL Vanillezucker}
  \ingredient{40\,g Speisestärke}
  verrühren, aufkochen, dabei ständig rühren.
  \ingredient[Mandel]{20\,g Mandelblätter, geröstet}
  und Eischnee unterziehen, in Schalen oder Gläser füllen.

  \ingredient[Weißwein]{200\,ml Weißwein}
  \ingredient{3 Eigelb}
  \ingredient{80\,g Zucker}
  \ingredient{1\,TL Speisestärke}
  \ingredient{\sfrac{1}{2} Zitrone (Saft, Schale)}
  in eine Rührschüssel geben, über Wasserdampf dickcremig aufschlagen.
  Auf die weisse Creme geben und erkalten lassen.

  Nach Wunsch mit Sahne verzieren.
\end{recipe}

  unterheben.
